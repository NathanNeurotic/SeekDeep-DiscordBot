# SeekDeep · boot deps — the minimum needed for local_ai_server.py to start,
# bind 7865, serve /gui/, and answer /health.
#
# This list is intentionally small (~30 MB on disk) so the Tauri shell's
# "first-run install" flow is fast and the unified .exe experience boots
# in under a minute.
#
# Heavy ML libraries (torch / transformers / diffusers / etc) live in
# requirements-ml.txt and are installed on first use of /chat / /image /
# /vision via the in-app "Install ML libraries" prompt. The server starts
# fine without them; ML-dependent handlers return ML_DEPS_MISSING until
# the user opts to install.

# Upper caps mirror requirements-ml.txt: this file IS installed by the Tauri
# first-run flow, so an unpinned `>=` would let a future breaking major (a
# pydantic 3, a fastapi/starlette major, etc.) silently land on a user's first
# launch with no CI gate. Caps are chosen to NOT exclude the known-good versions.
fastapi>=0.136.3,<1
uvicorn[standard]>=0.49.0,<1
pydantic>=2.10.0,<3
python-dotenv>=1.2.2,<2
pillow>=12.2.0,<13
requests>=2.34.2,<3
huggingface_hub>=0.30.0,<2
matplotlib>=3.9.0,<4
psutil>=7.2.2,<8

# Tokenizer backends. These are TINY (~10 MB combined) and pure-import —
# no GPU, no torch, no diffusers. Promoting them to boot deps because
# transformers needs at least one of them to load a fast tokenizer for
# every modern chat model (Llama 3.x → tiktoken, Llama 2/Mistral/Granite
# ≤3.2 → sentencepiece). Without them, /chat 503s on first load with
# "Couldn't instantiate the backend tokenizer". Boot-time install means
# no "click INSTALL TOKENIZER DEPS to continue" prompt on fresh installs.
tiktoken>=0.8.0,<1
sentencepiece>=0.2.1,<1
